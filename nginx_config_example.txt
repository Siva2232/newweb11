server {
    listen 80;
    server_name _;  # Or your domain name like example.com

    root /var/www/newwww/dist;  # Path to your React build
    index index.html;

    # Serve the React Frontend
    location / {
        try_files $uri $uri/ /index.html;
    }

    # Proxy API requests to the Backend (port 5000)
    location /api/ {
        proxy_pass http://localhost:5000/api/;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }

    # Serve uploaded images correctly
    location /uploads/ {
        alias /var/www/newweb11/uploads/;
    }
}
